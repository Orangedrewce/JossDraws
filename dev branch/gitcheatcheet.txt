# Git Cheat Sheet for Beginners

# WHERE AM I? - Check your current branch
git branch

# START WORK - Create a safe space for your changes
# This creates a new branch called 'dev' and switches to it.
# You only need to do this once for a new feature.
git checkout -b dev

# If 'dev' branch already exists, just switch to it
git checkout dev

# Get the latest changes from GitHub
git pull

# CHECK STATUS - See what you've changed
git status
git log --oneline

# IT WORKS - PROMOTE - Merge your changes into the main branch
# Switch to the main branch
git checkout main

# Merge the changes from your 'dev' branch
git merge dev

# Push your merged changes to GitHub
git push

# Switch back to your 'dev' branch for more work
git checkout dev

# IT'S BROKEN - DELETE - Start over with a clean slate
# Go back to the main branch
git checkout main

# Delete the broken 'dev' branch
git branch -D dev

# Create a fresh 'dev' branch from main
git checkout -b dev

# Force push the new 'dev' branch to GitHub (use with caution!)
git push -f origin dev

# UNDO RECENT MESS - Rewind to a previous good state
# Find the commit ID you want to go back to
git log --oneline

# Reset your branch to that commit (this will delete later changes!)
git reset --hard <id>

# Force push the reset branch to GitHub (use with caution!)
git push -f origin dev

# MATCH GITHUB EXACTLY - Make your local branch identical to the one on GitHub
# Get the latest info from GitHub
git fetch

# Force your local 'dev' branch to match the 'dev' branch on GitHub
git reset --hard origin/dev

# NUCLEAR RESET - Make your local branch identical to the main branch on GitHub
# Get the latest info from GitHub
git fetch

# Force your local branch to match the 'main' branch on GitHub (DANGEROUS!)
git reset --hard origin/main
